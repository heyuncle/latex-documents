    \begin{itemize}
        \item (hereditary property) if $A \subseteq B \in \mathcal{M}_I$, then $A \in \mathcal{M}_I$,
        \item (augmentation property) for $A, B \in \mathcal{M}_I$ with $|A| < |B|$ there exists $b \in B$ so that $A \cup \{b\} \in \mathcal{M}_I$.
    \end{itemize}
